017package org.apache.commons.math3.ode;
018
019import org.apache.commons.math3.exception.DimensionMismatchException;
020import org.apache.commons.math3.exception.MaxCountExceededException;
021
022/** Interface expanding {@link FirstOrderDifferentialEquations first order
023 *  differential equations} in order to compute exactly the main state jacobian
024 *  matrix for {@link JacobianMatrices partial derivatives equations}.
025 *
026 * @since 3.0
027 */
028public interface MainStateJacobianProvider extends FirstOrderDifferentialEquations {
029
030    /** Compute the jacobian matrix of ODE with respect to main state.
031     * @param t current value of the independent <I>time</I> variable
032     * @param y array containing the current value of the main state vector
033     * @param yDot array containing the current value of the time derivative of the main state vector
034     * @param dFdY placeholder array where to put the jacobian matrix of the ODE w.r.t. the main state vector
035     * @exception MaxCountExceededException if the number of functions evaluations is exceeded
036     * @exception DimensionMismatchException if arrays dimensions do not match equations settings
037     */
038    void computeMainStateJacobian(double t, double[] y, double[] yDot, double[][] dFdY)
039        throws MaxCountExceededException, DimensionMismatchException;
040
041}
