017package org.apache.commons.math3.optim.linear;
018
019/**
020 * Types of relationships between two cells in a Solver {@link LinearConstraint}.
021 *
022 * @since 2.0
023 */
024public enum Relationship {
025    /** Equality relationship. */
026    EQ("="),
027    /** Lesser than or equal relationship. */
028    LEQ("<="),
029    /** Greater than or equal relationship. */
030    GEQ(">=");
031
032    /** Display string for the relationship. */
033    private final String stringValue;
034
035    /**
036     * Simple constructor.
037     *
038     * @param stringValue Display string for the relationship.
039     */
040    Relationship(String stringValue) {
041        this.stringValue = stringValue;
042    }
043
044    /** {@inheritDoc} */
045    @Override
046    public String toString() {
047        return stringValue;
048    }
049
050    /**
051     * Gets the relationship obtained when multiplying all coefficients by -1.
052     *
053     * @return the opposite relationship.
054     */
055    public Relationship oppositeRelationship() {
056        switch (this) {
057        case LEQ :
058            return GEQ;
059        case GEQ :
060            return LEQ;
061        default :
062            return EQ;
063        }
064    }
065}
