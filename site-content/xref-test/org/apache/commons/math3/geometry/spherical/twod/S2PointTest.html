17  package org.apache.commons.math3.geometry.spherical.twod;
18  
19  import org.apache.commons.math3.exception.OutOfRangeException;
20  import org.apache.commons.math3.geometry.spherical.oned.Sphere1D;
21  import org.apache.commons.math3.util.FastMath;
22  import org.apache.commons.math3.util.MathUtils;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  public class S2PointTest {
27  
28  
29      @Test
30      public void testS2Point() {
31          for (int k = -2; k < 3; ++k) {
32              S2Point p = new S2Point(1.0 + k * MathUtils.TWO_PI, 1.4);
33              Assert.assertEquals(1.0 + k * MathUtils.TWO_PI, p.getTheta(), 1.0e-10);
34              Assert.assertEquals(1.4, p.getPhi(), 1.0e-10);
35              Assert.assertEquals(FastMath.cos(1.0) * FastMath.sin(1.4), p.getVector().getX(), 1.0e-10);
36              Assert.assertEquals(FastMath.sin(1.0) * FastMath.sin(1.4), p.getVector().getY(), 1.0e-10);
37              Assert.assertEquals(FastMath.cos(1.4), p.getVector().getZ(), 1.0e-10);
38              Assert.assertFalse(p.isNaN());
39          }
40      }
41  
42      @Test(expected=OutOfRangeException.class)
43      public void testNegativePolarAngle() {
44          new S2Point(1.0, -1.0);
45      }
46  
47      @Test(expected=OutOfRangeException.class)
48      public void testTooLargePolarAngle() {
49          new S2Point(1.0, 3.5);
50      }
51  
52      @Test
53      public void testNaN() {
54          Assert.assertTrue(S2Point.NaN.isNaN());
55          Assert.assertTrue(S2Point.NaN.equals(new S2Point(Double.NaN, 1.0)));
56          Assert.assertFalse(new S2Point(1.0, 1.3).equals(S2Point.NaN));
57      }
58  
59      @Test
60      public void testEquals() {
61          S2Point a = new S2Point(1.0, 1.0);
62          S2Point b = new S2Point(1.0, 1.0);
63          Assert.assertEquals(a.hashCode(), b.hashCode());
64          Assert.assertFalse(a == b);
65          Assert.assertTrue(a.equals(b));
66          Assert.assertTrue(a.equals(a));
67          Assert.assertFalse(a.equals('a'));
68      }
69  
70      @Test
71      public void testDistance() {
72          S2Point a = new S2Point(1.0, 0.5 * FastMath.PI);
73          S2Point b = new S2Point(a.getTheta() + 0.5 * FastMath.PI, a.getPhi());
74          Assert.assertEquals(0.5 * FastMath.PI, a.distance(b), 1.0e-10);
75          Assert.assertEquals(FastMath.PI, a.distance(a.negate()), 1.0e-10);
76          Assert.assertEquals(0.5 * FastMath.PI, S2Point.MINUS_I.distance(S2Point.MINUS_K), 1.0e-10);
77          Assert.assertEquals(0.0, new S2Point(1.0, 0).distance(new S2Point(2.0, 0)), 1.0e-10);
78      }
79  
80      @Test
81      public void testSpace() {
82          S2Point a = new S2Point(1.0, 1.0);
83          Assert.assertTrue(a.getSpace() instanceof Sphere2D);
84          Assert.assertEquals(2, a.getSpace().getDimension());
85          Assert.assertTrue(a.getSpace().getSubSpace() instanceof Sphere1D);
86      }

88  }
