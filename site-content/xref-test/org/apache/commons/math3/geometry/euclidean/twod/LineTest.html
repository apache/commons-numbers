32      @Test
33      public void testContains() {
34          Line l = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
35          Assert.assertTrue(l.contains(new Vector2D(0, 1)));
36          Assert.assertTrue(l.contains(new Vector2D(1, 2)));
37          Assert.assertTrue(l.contains(new Vector2D(7, 8)));
38          Assert.assertTrue(! l.contains(new Vector2D(8, 7)));
39      }

41      @Test
42      public void testAbscissa() {
43          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
44          Assert.assertEquals(0.0,
45                              (l.toSubSpace(new Vector2D(-3,  4))).getX(),
46                              1.0e-10);
47          Assert.assertEquals(0.0,
48                              (l.toSubSpace(new Vector2D( 3, -4))).getX(),
49                              1.0e-10);
50          Assert.assertEquals(-5.0,
51                              (l.toSubSpace(new Vector2D( 7, -1))).getX(),
52                              1.0e-10);
53          Assert.assertEquals( 5.0,
54                               (l.toSubSpace(new Vector2D(-1, -7))).getX(),
55                               1.0e-10);
56      }

58      @Test
59      public void testOffset() {
60          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
61          Assert.assertEquals(-5.0, l.getOffset(new Vector2D(5, -3)), 1.0e-10);
62          Assert.assertEquals(+5.0, l.getOffset(new Vector2D(-5, 2)), 1.0e-10);
63      }

65      @Test
66      public void testDistance() {
67          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
68          Assert.assertEquals(+5.0, l.distance(new Vector2D(5, -3)), 1.0e-10);
69          Assert.assertEquals(+5.0, l.distance(new Vector2D(-5, 2)), 1.0e-10);
70      }

72      @Test
73      public void testPointAt() {
74          Line l = new Line(new Vector2D(2, 1), new Vector2D(-2, -2), 1.0e-10);
75          for (double a = -2.0; a < 2.0; a += 0.2) {
76              Point<Euclidean1D> pA = new Vector1D(a);
77              Point<Euclidean2D> point = l.toSpace(pA);
78              Assert.assertEquals(a, (l.toSubSpace(point)).getX(), 1.0e-10);
79              Assert.assertEquals(0.0, l.getOffset(point),   1.0e-10);
80              for (double o = -2.0; o < 2.0; o += 0.2) {
81                  point = l.getPointAt((Vector1D) pA, o);
82                  Assert.assertEquals(a, (l.toSubSpace(point)).getX(), 1.0e-10);
83                  Assert.assertEquals(o, l.getOffset(point),   1.0e-10);
84              }
85          }
86      }

88      @Test
89      public void testOriginOffset() {
90          Line l1 = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
91          Assert.assertEquals(FastMath.sqrt(0.5), l1.getOriginOffset(), 1.0e-10);
92          Line l2 = new Line(new Vector2D(1, 2), new Vector2D(0, 1), 1.0e-10);
93          Assert.assertEquals(-FastMath.sqrt(0.5), l2.getOriginOffset(), 1.0e-10);
94      }

96      @Test
97      public void testParallel() {
98          Line l1 = new Line(new Vector2D(0, 1), new Vector2D(1, 2), 1.0e-10);
99          Line l2 = new Line(new Vector2D(2, 2), new Vector2D(3, 3), 1.0e-10);
100         Assert.assertTrue(l1.isParallelTo(l2));
101         Line l3 = new Line(new Vector2D(1, 0), new Vector2D(0.5, -0.5), 1.0e-10);
102         Assert.assertTrue(l1.isParallelTo(l3));
103         Line l4 = new Line(new Vector2D(1, 0), new Vector2D(0.5, -0.51), 1.0e-10);
104         Assert.assertTrue(! l1.isParallelTo(l4));
105     }

107     @Test
108     public void testTransform() throws MathIllegalArgumentException {
109 
110         Line l1 = new Line(new Vector2D(1.0 ,1.0), new Vector2D(4.0 ,1.0), 1.0e-10);
111         Transform<Euclidean2D, Euclidean1D> t1 =
112             Line.getTransform(0.0, 0.5, -1.0, 0.0, 1.0, 1.5);
113         Assert.assertEquals(0.5 * FastMath.PI,
114                             ((Line) t1.apply(l1)).getAngle(),
115                             1.0e-10);
116 
117         Line l2 = new Line(new Vector2D(0.0, 0.0), new Vector2D(1.0, 1.0), 1.0e-10);
118         Transform<Euclidean2D, Euclidean1D> t2 =
119             Line.getTransform(0.0, 0.5, -1.0, 0.0, 1.0, 1.5);
120         Assert.assertEquals(FastMath.atan2(1.0, -2.0),
121                             ((Line) t2.apply(l2)).getAngle(),
122                             1.0e-10);
123 
124     }

126     @Test
127     public void testIntersection() {
128         Line    l1 = new Line(new Vector2D( 0, 1), new Vector2D(1, 2), 1.0e-10);
129         Line    l2 = new Line(new Vector2D(-1, 2), new Vector2D(2, 1), 1.0e-10);
130         Vector2D p  = l1.intersection(l2);
131         Assert.assertEquals(0.5, p.getX(), 1.0e-10);
132         Assert.assertEquals(1.5, p.getY(), 1.0e-10);
133     }
