1033 
1034         double eps      = 5.0e-16;
1035         double kx       = 2;
1036         double ky       = -3;
1037         double kz       = 5;
1038         double n2       = kx * kx + ky * ky + kz * kz;
1039         double n        = FastMath.sqrt(n2);
1040         double theta    = 1.7;
1041         double cosTheta = FastMath.cos(theta);
1042         double sinTheta = FastMath.sin(theta);
1043         FieldRotation<DerivativeStructure> r    = new FieldRotation<DerivativeStructure>(createAxis(kx, ky, kz),
1044                                                                                          createAngle(theta),
1045                                                                                          RotationConvention.VECTOR_OPERATOR);
1046         Vector3D a      = new Vector3D(kx / n, ky / n, kz / n);
1047 
1048         // Jacobian of the normalized rotation axis a with respect to the Cartesian vector k
1049         RealMatrix dadk = MatrixUtils.createRealMatrix(new double[][] {
1050             { (ky * ky + kz * kz) / ( n * n2),            -kx * ky / ( n * n2),            -kx * kz / ( n * n2) },
1051             {            -kx * ky / ( n * n2), (kx * kx + kz * kz) / ( n * n2),            -ky * kz / ( n * n2) },
1052             {            -kx * kz / ( n * n2),            -ky * kz / ( n * n2), (kx * kx + ky * ky) / ( n * n2) }
1053         });
1054 
1055         for (double x = -0.9; x < 0.9; x += 0.2) {
1056             for (double y = -0.9; y < 0.9; y += 0.2) {
1057                 for (double z = -0.9; z < 0.9; z += 0.2) {
1058                     Vector3D   u = new Vector3D(x, y, z);
1059                     FieldVector3D<DerivativeStructure> v = r.applyTo(createVector(x, y, z));
1060 
1061                     // explicit formula for rotation of vector u around axis a with angle theta
1062                     double dot     = Vector3D.dotProduct(u, a);
1063                     Vector3D cross = Vector3D.crossProduct(a, u);
1064                     double c1      = 1 - cosTheta;
1065                     double c2      = c1 * dot;
1066                     Vector3D rt    = new Vector3D(cosTheta, u, c2, a, sinTheta, cross);
1067                     Assert.assertEquals(rt.getX(), v.getX().getReal(), eps);
1068                     Assert.assertEquals(rt.getY(), v.getY().getReal(), eps);
1069                     Assert.assertEquals(rt.getZ(), v.getZ().getReal(), eps);
1070 
1071                     // Jacobian of the image v = r(u) with respect to rotation axis a
1072                     // (analytical differentiation of the explicit formula)
1073                     RealMatrix dvda = MatrixUtils.createRealMatrix(new double[][] {
1074                         { c1 * x * a.getX() + c2,           c1 * y * a.getX() + sinTheta * z, c1 * z * a.getX() - sinTheta * y },
1075                         { c1 * x * a.getY() - sinTheta * z, c1 * y * a.getY() + c2,           c1 * z * a.getY() + sinTheta * x },
1076                         { c1 * x * a.getZ() + sinTheta * y, c1 * y * a.getZ() - sinTheta * x, c1 * z * a.getZ() + c2           }
1077                     });
1078 
1079                     // compose Jacobians
1080                     RealMatrix dvdk = dvda.multiply(dadk);
1081 
1082                     // derivatives with respect to un-normalized axis
1083                     Assert.assertEquals(dvdk.getEntry(0, 0), v.getX().getPartialDerivative(1, 0, 0, 0), eps);
1084                     Assert.assertEquals(dvdk.getEntry(0, 1), v.getX().getPartialDerivative(0, 1, 0, 0), eps);
