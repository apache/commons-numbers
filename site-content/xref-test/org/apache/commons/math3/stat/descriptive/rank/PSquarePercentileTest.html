63       * Verifies that copied statistics remain equal to originals when
64       * incremented the same way by making the copy after a majority of elements
65       * are incremented
66       */
67      @Test
68      public void testCopyConsistencyWithInitialMostElements() {
69  
70          StorelessUnivariateStatistic master =
71                  (StorelessUnivariateStatistic) getUnivariateStatistic();
72  
73          StorelessUnivariateStatistic replica = null;
74  
75          // select a portion of testArray till 75 % of the length to load first
76          long index = FastMath.round(0.75 * testArray.length);
77  
78          // Put first half in master and copy master to replica
79          master.incrementAll(testArray, 0, (int) index);
80          replica = master.copy();
81  
82          // Check same
83          Assert.assertTrue(replica.equals(master));
84          Assert.assertTrue(master.equals(replica));
85  
86          // Now add second part to both and check again
87          master.incrementAll(testArray, (int) index,
88                  (int) (testArray.length - index));
89          replica.incrementAll(testArray, (int) index,
90                  (int) (testArray.length - index));
91          Assert.assertTrue(replica.equals(master));
92          Assert.assertTrue(master.equals(replica));
93      }

540     private void assertValues(Double a, Double b, double delta) {
541         if (Double.isNaN(a)) {
542             Assert.assertTrue("" + b + " is not NaN.", Double.isNaN(a));
543         } else {
544             double max = FastMath.max(a, b);
545             double percentage = FastMath.abs(a - b) / max;
546             double deviation = delta;
547             Assert.assertTrue(String.format(
548                     "Deviated = %f and is beyond %f as a=%f,  b=%f",
549                     percentage, deviation, a, b), percentage < deviation);
550         }
551     }

553     private void doCalculatePercentile(Double percentile, Number[] test) {
554         doCalculatePercentile(percentile, test, Double.MAX_VALUE);
555     }

557     private void doCalculatePercentile(Double percentile, Number[] test,
558             double delta) {
559         PSquarePercentile psquared = new PSquarePercentile(percentile);
560         for (Number value : test) {
561             psquared.increment(value.doubleValue());
562         }
563 
564         Percentile p2 = new Percentile(percentile * 100);
565 
566         double[] dall = new double[test.length];
567         for (int i = 0; i < test.length; i++) {
568             dall[i] = test[i].doubleValue();
569         }
570 
571         Double referenceValue = p2.evaluate(dall);
572         assertValues(psquared.getResult(), referenceValue, delta);
573     }
574 
575     private void doCalculatePercentile(double percentile, double[] test,
576             double delta) {
577         PSquarePercentile psquared = new PSquarePercentile(percentile);
578         for (double value : test) {
579             psquared.increment(value);
580         }
581 
582         Percentile p2 =
583                 new Percentile(percentile < 1 ? percentile * 100 : percentile);
584         /*
585          * double[] dall = new double[test.length]; for (int i = 0; i <
586          * test.length; i++) dall[i] = test[i];
587          */
588         Double referenceValue = p2.evaluate(test);
589         assertValues(psquared.getResult(), referenceValue, delta);
590     }
