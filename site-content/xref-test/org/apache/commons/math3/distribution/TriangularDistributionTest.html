31      // --- Override tolerance -------------------------------------------------
32  
33      @Override
34      public void setUp() {
35          super.setUp();
36          setTolerance(1e-4);
37      }

39      //--- Implementations for abstract methods --------------------------------

41      /**
42       * Creates the default triangular distribution instance to use in tests.
43       */
44      @Override
45      public TriangularDistribution makeDistribution() {
46          // Left side 5 wide, right side 10 wide.
47          return new TriangularDistribution(-3, 2, 12);
48      }
49  
50      /**
51       * Creates the default cumulative probability distribution test input
52       * values.
53       */
54      @Override
55      public double[] makeCumulativeTestPoints() {
56          return new double[] { -3.0001,                 // below lower limit
57                                -3.0,                    // at lower limit
58                                -2.0, -1.0, 0.0, 1.0,    // on lower side
59                                2.0,                     // at mode
60                                3.0, 4.0, 10.0, 11.0,    // on upper side
61                                12.0,                    // at upper limit
62                                12.0001                  // above upper limit
63                              };
64      }
65  
66      /**
67       * Creates the default cumulative probability density test expected values.
68       */
69      @Override
70      public double[] makeCumulativeTestValues() {
71          // Top at 2 / (b - a) = 2 / (12 - -3) = 2 / 15 = 7.5
72          // Area left  = 7.5 * 5  * 0.5 = 18.75 (1/3 of the total area)
73          // Area right = 7.5 * 10 * 0.5 = 37.5  (2/3 of the total area)
74          // Area total = 18.75 + 37.5 = 56.25
75          // Derivative left side = 7.5 / 5 = 1.5
76          // Derivative right side = -7.5 / 10 = -0.75
77          double third = 1 / 3.0;
78          double left = 18.75;
79          double area = 56.25;
80          return new double[] { 0.0,
81                                0.0,
82                                0.75 / area, 3 / area, 6.75 / area, 12 / area,
83                                third,
84                                (left + 7.125) / area, (left + 13.5) / area,
85                                (left + 36) / area, (left + 37.125) / area,
86                                1.0,
87                                1.0
88                              };
89      }
90  
91      /**
92       * Creates the default inverse cumulative probability distribution test
93       * input values.
94       */
95      @Override
96      public double[] makeInverseCumulativeTestPoints() {
97          // Exclude the points outside the limits, as they have cumulative
98          // probability of zero and one, meaning the inverse returns the
99          // limits and not the points outside the limits.
100         double[] points = makeCumulativeTestValues();
101         double[] points2 = new double[points.length-2];
102         System.arraycopy(points, 1, points2, 0, points2.length);
103         return points2;
104         //return Arrays.copyOfRange(points, 1, points.length - 1);
105     }
106

138     /** Test lower bound getter. */
139     @Test
140     public void testGetLowerBound() {
141         TriangularDistribution distribution = makeDistribution();
142         Assert.assertEquals(-3.0, distribution.getSupportLowerBound(), 0);
143     }

145     /** Test upper bound getter. */
146     @Test
147     public void testGetUpperBound() {

150     }

152     /** Test pre-condition for equal lower/upper limit. */
153     @Test(expected=NumberIsTooLargeException.class)
154     public void testPreconditions1() {
155         new TriangularDistribution(0, 0, 0);
156     }
157 
158     /** Test pre-condition for lower limit larger than upper limit. */
159     @Test(expected=NumberIsTooLargeException.class)

161         new TriangularDistribution(1, 1, 0);

164     /** Test pre-condition for mode larger than upper limit. */
165     @Test(expected=NumberIsTooLargeException.class)

170     /** Test pre-condition for mode smaller than lower limit. */
171     @Test(expected=NumberIsTooSmallException.class)
172     public void testPreconditions4() {
173         new TriangularDistribution(2, 1, 3);
174     }
175 
176     /** Test mean/variance. */
177     @Test
178     public void testMeanVariance() {
179         TriangularDistribution dist;
180 
181         dist = new TriangularDistribution(0, 0.5, 1.0);
182         Assert.assertEquals(dist.getNumericalMean(), 0.5, 0);
183         Assert.assertEquals(dist.getNumericalVariance(), 1 / 24.0, 0);
184 
185         dist = new TriangularDistribution(0, 1, 1);
186         Assert.assertEquals(dist.getNumericalMean(), 2 / 3.0, 0);
187         Assert.assertEquals(dist.getNumericalVariance(), 1 / 18.0, 0);
188 
189         dist = new TriangularDistribution(-3, 2, 12);
190         Assert.assertEquals(dist.getNumericalMean(), 3 + (2 / 3.0), 0);
191         Assert.assertEquals(dist.getNumericalVariance(), 175 / 18.0, 0);
192     }
