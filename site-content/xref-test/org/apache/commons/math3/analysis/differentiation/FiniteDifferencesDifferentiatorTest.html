63                  new FiniteDifferencesDifferentiator(5, 0.01);

65                  differentiator.differentiate(new UnivariateFunction() {
66                      public double value(double x) {

96      @Test
97      public void testGaussian() {
98          FiniteDifferencesDifferentiator differentiator =
99                  new FiniteDifferencesDifferentiator(9, 0.02);
100         UnivariateDifferentiableFunction gaussian = new Gaussian(1.0, 2.0);
101         UnivariateDifferentiableFunction f =
102                 differentiator.differentiate(gaussian);
103         double[] expectedError = new double[] {
104             6.939e-18, 1.284e-15, 2.477e-13, 1.168e-11, 2.840e-9, 7.971e-8
105         };
106        double[] maxError = new double[expectedError.length];
107         for (double x = -10; x < 10; x += 0.1) {
108             DerivativeStructure dsX  = new DerivativeStructure(1, maxError.length - 1, 0, x);
109             DerivativeStructure yRef = gaussian.value(dsX);
110             DerivativeStructure y    = f.value(dsX);
111             Assert.assertEquals(f.value(dsX.getValue()), f.value(dsX).getValue(), 1.0e-15);
112             for (int order = 0; order <= yRef.getOrder(); ++order) {
113                 maxError[order] = FastMath.max(maxError[order],
114                                         FastMath.abs(yRef.getPartialDerivative(order) -
115                                                      y.getPartialDerivative(order)));
116             }
117         }
118         for (int i = 0; i < maxError.length; ++i) {
119             Assert.assertEquals(expectedError[i], maxError[i], 0.01 * expectedError[i]);
120         }
121     }

123     @Test
124     public void testStepSizeUnstability() {
125         UnivariateDifferentiableFunction quintic = new QuinticFunction();
126         UnivariateDifferentiableFunction goodStep =
127                 new FiniteDifferencesDifferentiator(7, 0.25).differentiate(quintic);
128         UnivariateDifferentiableFunction badStep =
129                 new FiniteDifferencesDifferentiator(7, 1.0e-6).differentiate(quintic);
130         double[] maxErrorGood = new double[7];
131         double[] maxErrorBad  = new double[7];
132         for (double x = -10; x < 10; x += 0.1) {
133             DerivativeStructure dsX  = new DerivativeStructure(1, 6, 0, x);
134             DerivativeStructure yRef  = quintic.value(dsX);
135             DerivativeStructure yGood = goodStep.value(dsX);
136             DerivativeStructure yBad  = badStep.value(dsX);
137             for (int order = 0; order <= 6; ++order) {
138                 maxErrorGood[order] = FastMath.max(maxErrorGood[order],
139                                                    FastMath.abs(yRef.getPartialDerivative(order) -
140                                                                 yGood.getPartialDerivative(order)));
141                 maxErrorBad[order]  = FastMath.max(maxErrorBad[order],
142                                                    FastMath.abs(yRef.getPartialDerivative(order) -
143                                                                 yBad.getPartialDerivative(order)));
144             }
145         }
146 
147         // the 0.25 step size is good for finite differences in the quintic on this abscissa range for 7 points
148         // the errors are fair
149         final double[] expectedGood = new double[] {
150             7.276e-12, 7.276e-11, 9.968e-10, 3.092e-9, 5.432e-8, 8.196e-8, 1.818e-6
151         };
152 
153         // the 1.0e-6 step size is far too small for finite differences in the quintic on this abscissa range for 7 points
154         // the errors are huge!
155         final double[] expectedBad = new double[] {
156             2.910e-11, 2.087e-5, 147.7, 3.820e7, 6.354e14, 6.548e19, 1.543e27
157         };
158 
159         for (int i = 0; i < maxErrorGood.length; ++i) {
160             Assert.assertEquals(expectedGood[i], maxErrorGood[i], 0.01 * expectedGood[i]);
161             Assert.assertEquals(expectedBad[i],  maxErrorBad[i],  0.01 * expectedBad[i]);
162         }
163 
164     }

210     @Test
211     public void testBounds() {
212 
213         final double slope = 2.5;
214         UnivariateFunction f = new UnivariateFunction() {
215             public double value(double x) {
216                 if (x < 0) {
217                     throw new NumberIsTooSmallException(x, 0, true);
218                 } else if (x > 1) {
219                     throw new NumberIsTooLargeException(x, 1, true);
220                 } else {
221                     return slope * x;
222                 }
223             }
224         };
225 
226         UnivariateDifferentiableFunction missingBounds =
227                 new FiniteDifferencesDifferentiator(3, 0.1).differentiate(f);
228         UnivariateDifferentiableFunction properlyBounded =
229                 new FiniteDifferencesDifferentiator(3, 0.1, 0.0, 1.0).differentiate(f);
230         DerivativeStructure tLow  = new DerivativeStructure(1, 1, 0, 0.05);
231         DerivativeStructure tHigh = new DerivativeStructure(1, 1, 0, 0.95);
232 
233         try {
234             // here, we did not set the bounds, so the differences are evaluated out of domain
235             // using f(-0.05), f(0.05), f(0.15)
236             missingBounds.value(tLow);
237             Assert.fail("an exception should have been thrown");
238         } catch (NumberIsTooSmallException nse) {
239             Assert.assertEquals(-0.05, nse.getArgument().doubleValue(), 1.0e-10);
240         } catch (Exception e) {
241             Assert.fail("wrong exception caught: " + e.getClass().getName());
242         }
243 
244         try {
245             // here, we did not set the bounds, so the differences are evaluated out of domain
246             // using f(0.85), f(0.95), f(1.05)
247             missingBounds.value(tHigh);
248             Assert.fail("an exception should have been thrown");
249         } catch (NumberIsTooLargeException nle) {
250             Assert.assertEquals(1.05, nle.getArgument().doubleValue(), 1.0e-10);
251         } catch (Exception e) {
252             Assert.fail("wrong exception caught: " + e.getClass().getName());
253         }
254 
255         // here, we did set the bounds, so evaluations are done within domain
256         // using f(0.0), f(0.1), f(0.2)
257         Assert.assertEquals(slope, properlyBounded.value(tLow).getPartialDerivative(1), 1.0e-10);
258 
259         // here, we did set the bounds, so evaluations are done within domain
260         // using f(0.8), f(0.9), f(1.0)
261         Assert.assertEquals(slope, properlyBounded.value(tHigh).getPartialDerivative(1), 1.0e-10);
262 
263     }

265     @Test
266     public void testBoundedSqrt() {
267 
268         UnivariateFunctionDifferentiator differentiator =
269                 new FiniteDifferencesDifferentiator(9, 1.0 / 32, 0.0, Double.POSITIVE_INFINITY);
270         UnivariateDifferentiableFunction sqrt = differentiator.differentiate(new UnivariateFunction() {
271             public double value(double x) {
272                 return FastMath.sqrt(x);
273             }
274         });
275 
276         // we are able to compute derivative near 0, but the accuracy is much poorer there
277         DerivativeStructure t001 = new DerivativeStructure(1, 1, 0, 0.01);
278         Assert.assertEquals(0.5 / FastMath.sqrt(t001.getValue()), sqrt.value(t001).getPartialDerivative(1), 1.6);
279         DerivativeStructure t01 = new DerivativeStructure(1, 1, 0, 0.1);
280         Assert.assertEquals(0.5 / FastMath.sqrt(t01.getValue()), sqrt.value(t01).getPartialDerivative(1), 7.0e-3);
281         DerivativeStructure t03 = new DerivativeStructure(1, 1, 0, 0.3);
282         Assert.assertEquals(0.5 / FastMath.sqrt(t03.getValue()), sqrt.value(t03).getPartialDerivative(1), 2.1e-7);
283 
284     }

286     @Test
287     public void testVectorFunction() {
288 
289         FiniteDifferencesDifferentiator differentiator =
290                 new FiniteDifferencesDifferentiator(7, 0.01);
291         UnivariateDifferentiableVectorFunction f =
292                 differentiator.differentiate(new UnivariateVectorFunction() {
293 
294             public double[] value(double x) {
295                 return new double[] { FastMath.cos(x), FastMath.sin(x) };
296             }
297 
298         });
299 
300         for (double x = -10; x < 10; x += 0.1) {
301             DerivativeStructure dsX = new DerivativeStructure(1, 2, 0, x);
302             DerivativeStructure[] y = f.value(dsX);
303             double cos = FastMath.cos(x);
304             double sin = FastMath.sin(x);
305             double[] f1 = f.value(dsX.getValue());
306             DerivativeStructure[] f2 = f.value(dsX);
307             Assert.assertEquals(f1.length, f2.length);
308             for (int i = 0; i < f1.length; ++i) {
309                 Assert.assertEquals(f1[i], f2[i].getValue(), 1.0e-15);
310             }
311             Assert.assertEquals( cos, y[0].getValue(), 7.0e-16);
312             Assert.assertEquals( sin, y[1].getValue(), 7.0e-16);
313             Assert.assertEquals(-sin, y[0].getPartialDerivative(1), 6.0e-14);
314             Assert.assertEquals( cos, y[1].getPartialDerivative(1), 6.0e-14);
315             Assert.assertEquals(-cos, y[0].getPartialDerivative(2), 2.0e-11);
316             Assert.assertEquals(-sin, y[1].getPartialDerivative(2), 2.0e-11);
317         }

321     @Test
322     public void testMatrixFunction() {
323 
324         FiniteDifferencesDifferentiator differentiator =
325                 new FiniteDifferencesDifferentiator(7, 0.01);
326         UnivariateDifferentiableMatrixFunction f =
327                 differentiator.differentiate(new UnivariateMatrixFunction() {
328 
329             public double[][] value(double x) {
330                 return new double[][] {
331                     { FastMath.cos(x),  FastMath.sin(x)  },
332                     { FastMath.cosh(x), FastMath.sinh(x) }
333                 };
334             }
335 
336         });
337 
338         for (double x = -1; x < 1; x += 0.02) {
339             DerivativeStructure dsX = new DerivativeStructure(1, 2, 0, x);
340             DerivativeStructure[][] y = f.value(dsX);
341             double cos = FastMath.cos(x);
342             double sin = FastMath.sin(x);
343             double cosh = FastMath.cosh(x);
344             double sinh = FastMath.sinh(x);
345             double[][] f1 = f.value(dsX.getValue());
346             DerivativeStructure[][] f2 = f.value(dsX);
347             Assert.assertEquals(f1.length, f2.length);
348             for (int i = 0; i < f1.length; ++i) {
349                 Assert.assertEquals(f1[i].length, f2[i].length);
350                 for (int j = 0; j < f1[i].length; ++j) {
351                     Assert.assertEquals(f1[i][j], f2[i][j].getValue(), 1.0e-15);
352                 }
353             }
354             Assert.assertEquals(cos,   y[0][0].getValue(), 7.0e-18);
355             Assert.assertEquals(sin,   y[0][1].getValue(), 6.0e-17);
356             Assert.assertEquals(cosh,  y[1][0].getValue(), 3.0e-16);
357             Assert.assertEquals(sinh,  y[1][1].getValue(), 3.0e-16);
358             Assert.assertEquals(-sin,  y[0][0].getPartialDerivative(1), 2.0e-14);
359             Assert.assertEquals( cos,  y[0][1].getPartialDerivative(1), 2.0e-14);
360             Assert.assertEquals( sinh, y[1][0].getPartialDerivative(1), 3.0e-14);
361             Assert.assertEquals( cosh, y[1][1].getPartialDerivative(1), 3.0e-14);
362             Assert.assertEquals(-cos,  y[0][0].getPartialDerivative(2), 3.0e-12);
363             Assert.assertEquals(-sin,  y[0][1].getPartialDerivative(2), 3.0e-12);
364             Assert.assertEquals( cosh, y[1][0].getPartialDerivative(2), 6.0e-12);
365             Assert.assertEquals( sinh, y[1][1].getPartialDerivative(2), 6.0e-12);
366         }

370     @Test
371     public void testSeveralFreeParameters() {
372         FiniteDifferencesDifferentiator differentiator =
373                 new FiniteDifferencesDifferentiator(5, 0.001);
374         UnivariateDifferentiableFunction sine = new Sin();
375         UnivariateDifferentiableFunction f =
376                 differentiator.differentiate(sine);
377         double[] expectedError = new double[] {
378             6.696e-16, 1.371e-12, 2.007e-8, 1.754e-5
379         };
380         double[] maxError = new double[expectedError.length];
381        for (double x = -2; x < 2; x += 0.1) {
382            for (double y = -2; y < 2; y += 0.1) {
383                DerivativeStructure dsX  = new DerivativeStructure(2, maxError.length - 1, 0, x);
384                DerivativeStructure dsY  = new DerivativeStructure(2, maxError.length - 1, 1, y);
385                DerivativeStructure dsT  = dsX.multiply(3).subtract(dsY.multiply(2));
386                DerivativeStructure sRef = sine.value(dsT);
387                DerivativeStructure s    = f.value(dsT);
388                for (int xOrder = 0; xOrder <= sRef.getOrder(); ++xOrder) {
389                    for (int yOrder = 0; yOrder <= sRef.getOrder(); ++yOrder) {
390                        if (xOrder + yOrder <= sRef.getOrder()) {
391                            maxError[xOrder +yOrder] = FastMath.max(maxError[xOrder + yOrder],
392                                                                     FastMath.abs(sRef.getPartialDerivative(xOrder, yOrder) -
393                                                                                  s.getPartialDerivative(xOrder, yOrder)));
394                        }
395                    }
396                }
397            }
398        }
399        for (int i = 0; i < maxError.length; ++i) {
400            Assert.assertEquals(expectedError[i], maxError[i], 0.01 * expectedError[i]);
401        }
402     }
