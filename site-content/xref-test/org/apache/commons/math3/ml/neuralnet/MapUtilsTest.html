20  import java.util.Set;

26  import org.junit.Assert;

28  /**
29   * Tests for {@link MapUtils} class.
30   */
31  public class MapUtilsTest {
32      /*
33       * Test assumes that the network is
34       *
35       *  0-----1-----2
36       */
37      @Test
38      public void testFindClosestNeuron() {
39          final FeatureInitializer init
40              = new OffsetFeatureInitializer(FeatureInitializerFactory.uniform(-0.1, 0.1));
41          final FeatureInitializer[] initArray = { init };
42  
43          final Network net = new NeuronString(3, false, initArray).getNetwork();
44          final DistanceMeasure dist = new EuclideanDistance();
45  
46          final Set<Neuron> allBest = new HashSet<Neuron>();
47          final Set<Neuron> best = new HashSet<Neuron>();
48          double[][] features;
49  
50          // The following tests ensures that
51          // 1. the same neuron is always selected when the input feature is
52          //    in the range of the initializer,
53          // 2. different network's neuron have been selected by inputs features
54          //    that belong to different ranges.
55  
56          best.clear();
57          features = new double[][] {
58              { -1 },
59              { 0.4 },
60          };
61          for (double[] f : features) {
62              best.add(MapUtils.findBest(f, net, dist));
63          }
64          Assert.assertEquals(1, best.size());
65          allBest.addAll(best);
66  
67          best.clear();
68          features = new double[][] {
69              { 0.6 },
70              { 1.4 },
71          };
72          for (double[] f : features) {
73              best.add(MapUtils.findBest(f, net, dist));
74          }
75          Assert.assertEquals(1, best.size());
76          allBest.addAll(best);
77  
78          best.clear();
79          features = new double[][] {
80              { 1.6 },
81              { 3 },
82          };
83          for (double[] f : features) {
84              best.add(MapUtils.findBest(f, net, dist));
85          }
86          Assert.assertEquals(1, best.size());
87          allBest.addAll(best);
88  
89          Assert.assertEquals(3, allBest.size());
90      }
91  
92      @Test
93      public void testSort() {
94          final Set<Neuron> list = new HashSet<Neuron>();
95  
96          for (int i = 0; i < 4; i++) {
97              list.add(new Neuron(i, new double[] { i - 0.5 }));
98          }
99  
100         final Neuron[] sorted = MapUtils.sort(new double[] { 3.4 },
101                                               list,
102                                               new EuclideanDistance());
103 
104         final long[] expected = new long[] { 3, 2, 1, 0 };
105         for (int i = 0; i < list.size(); i++) {
106             Assert.assertEquals(expected[i], sorted[i].getIdentifier());
107         }
108     }
109 }
