68      @Test
69      public void testMath372() {
70          LevenbergMarquardtOptimizer optimizer = new LevenbergMarquardtOptimizer();
71          CurveFitter<ParametricUnivariateFunction> curveFitter = new CurveFitter<ParametricUnivariateFunction>(optimizer);
72  
73          curveFitter.addObservedPoint( 15,  4443);
74          curveFitter.addObservedPoint( 31,  8493);
75          curveFitter.addObservedPoint( 62, 17586);
76          curveFitter.addObservedPoint(125, 30582);
77          curveFitter.addObservedPoint(250, 45087);
78          curveFitter.addObservedPoint(500, 50683);
79  
80          ParametricUnivariateFunction f = new ParametricUnivariateFunction() {
81              public double value(double x, double ... parameters) {
82                  double a = parameters[0];
83                  double b = parameters[1];
84                  double c = parameters[2];
85                  double d = parameters[3];
86  
87                  return d + ((a - d) / (1 + FastMath.pow(x / c, b)));
88              }
89  
90              public double[] gradient(double x, double ... parameters) {
91                  double a = parameters[0];
92                  double b = parameters[1];
93                  double c = parameters[2];
94                  double d = parameters[3];
95  
96                  double[] gradients = new double[4];
97                  double den = 1 + FastMath.pow(x / c, b);
98  
99                  // derivative with respect to a
100                 gradients[0] = 1 / den;
101 
102                 // derivative with respect to b
103                 // in the reported (invalid) issue, there was a sign error here
104                 gradients[1] = -((a - d) * FastMath.pow(x / c, b) * FastMath.log(x / c)) / (den * den);
105 
106                 // derivative with respect to c
107                 gradients[2] = (b * FastMath.pow(x / c, b - 1) * (x / (c * c)) * (a - d)) / (den * den);
108 
109                 // derivative with respect to d
110                 gradients[3] = 1 - (1 / den);
111 
112                 return gradients;
113 
114             }
115         };
116 
117         double[] initialGuess = new double[] { 1500, 0.95, 65, 35000 };
118         double[] estimatedParameters = curveFitter.fit(f, initialGuess);
119 
120         Assert.assertEquals( 2411.00, estimatedParameters[0], 500.00);
121         Assert.assertEquals(    1.62, estimatedParameters[1],   0.04);
122         Assert.assertEquals(  111.22, estimatedParameters[2],   0.30);
123         Assert.assertEquals(55347.47, estimatedParameters[3], 300.00);
124         Assert.assertTrue(optimizer.getRMS() < 600.0);
125     }
126 
127     private static class SimpleInverseFunction implements ParametricUnivariateFunction {
128 
129         public double value(double x, double ... parameters) {
130             return parameters[0] / x + (parameters.length < 2 ? 0 : parameters[1]);
131         }
132 
133         public double[] gradient(double x, double ... doubles) {
134             double[] gradientVector = new double[doubles.length];
135             gradientVector[0] = 1 / x;
136             if (doubles.length >= 2) {
137                 gradientVector[1] = 1;
138             }
139             return gradientVector;
140         }
141     }
