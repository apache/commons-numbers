017package org.apache.commons.math3.fitting.leastsquares;
018
019import java.io.BufferedReader;
020import java.io.IOException;
021import java.util.ArrayList;
022import org.apache.commons.math3.analysis.MultivariateVectorFunction;
023import org.apache.commons.math3.analysis.MultivariateMatrixFunction;
024import org.apache.commons.math3.util.MathArrays;

026/**
027 * This class gives access to the statistical reference datasets provided by the
028 * NIST (available
029 * <a href="http://www.itl.nist.gov/div898/strd/general/dataarchive.html">here</a>).
030 * Instances of this class can be created by invocation of the
031 * {@link StatisticalReferenceDatasetFactory}.
032 */
033public abstract class StatisticalReferenceDataset {
034    /** The name of this dataset. */
035    private final String name;
036    /** The total number of observations (data points). */
037    private final int numObservations;
038    /** The total number of parameters. */
039    private final int numParameters;
040    /** The total number of starting points for the optimizations. */
041    private final int numStartingPoints;
042    /** The values of the predictor. */
043    private final double[] x;
044    /** The values of the response. */
045    private final double[] y;
046    /**
047     * The starting values. {@code startingValues[j][i]} is the value of the
048     * {@code i}-th parameter in the {@code j}-th set of starting values.
049     */
050    private final double[][] startingValues;
051    /** The certified values of the parameters. */
052    private final double[] a;
053    /** The certified values of the standard deviation of the parameters. */
054    private final double[] sigA;
055    /** The certified value of the residual sum of squares. */
056    private double residualSumOfSquares;
057    /** The least-squares problem. */
058    private final LeastSquaresProblem problem;
059
060    /**
061     * Creates a new instance of this class from the specified data file. The
062     * file must follow the StRD format.
063     *
064     * @param in the data file
065     * @throws IOException if an I/O error occurs
066     */
067    public StatisticalReferenceDataset(final BufferedReader in)
068        throws IOException {
069
070        final ArrayList<String> lines = new ArrayList<String>();
071        for (String line = in.readLine(); line != null; line = in.readLine()) {
072            lines.add(line);
073        }
074        int[] index = findLineNumbers("Data", lines);
075        if (index == null) {
076            throw new AssertionError("could not find line indices for data");
077        }
078        this.numObservations = index[1] - index[0] + 1;
079        this.x = new double[this.numObservations];
080        this.y = new double[this.numObservations];
081        for (int i = 0; i < this.numObservations; i++) {
082            final String line = lines.get(index[0] + i - 1);
083            final String[] tokens = line.trim().split(" ++");
084            // Data columns are in reverse order!!!
085            this.y[i] = Double.parseDouble(tokens[0]);
086            this.x[i] = Double.parseDouble(tokens[1]);
087        }
088
089        index = findLineNumbers("Starting Values", lines);
090        if (index == null) {
091            throw new AssertionError(
092                                     "could not find line indices for starting values");
093        }
094        this.numParameters = index[1] - index[0] + 1;
095
096        double[][] start = null;
097        this.a = new double[numParameters];
098        this.sigA = new double[numParameters];
099        for (int i = 0; i < numParameters; i++) {
100            final String line = lines.get(index[0] + i - 1);
101            final String[] tokens = line.trim().split(" ++");
102            if (start == null) {
103                start = new double[tokens.length - 4][numParameters];
104            }
105            for (int j = 2; j < tokens.length - 2; j++) {
106                start[j - 2][i] = Double.parseDouble(tokens[j]);
107            }
108            this.a[i] = Double.parseDouble(tokens[tokens.length - 2]);
109            this.sigA[i] = Double.parseDouble(tokens[tokens.length - 1]);
110        }
111        if (start == null) {
112            throw new IOException("could not find starting values");
113        }
114        this.numStartingPoints = start.length;
115        this.startingValues = start;
116
117        double dummyDouble = Double.NaN;
118        String dummyString = null;
119        for (String line : lines) {
120            if (line.contains("Dataset Name:")) {
121                dummyString = line
122                    .substring(line.indexOf("Dataset Name:") + 13,
123                               line.indexOf("(")).trim();
124            }
125            if (line.contains("Residual Sum of Squares")) {
126                final String[] tokens = line.split(" ++");
127                dummyDouble = Double.parseDouble(tokens[4].trim());
128            }
129        }
130        if (Double.isNaN(dummyDouble)) {
131            throw new IOException(
132                                  "could not find certified value of residual sum of squares");
133        }
134        this.residualSumOfSquares = dummyDouble;
135
136        if (dummyString == null) {
137            throw new IOException("could not find dataset name");
138        }
139        this.name = dummyString;
140
141        this.problem = new LeastSquaresProblem();
142    }
143
144    class LeastSquaresProblem {
145        public MultivariateVectorFunction getModelFunction() {
146            return new MultivariateVectorFunction() {
147                public double[] value(final double[] a) {
148                    final int n = getNumObservations();
149                    final double[] yhat = new double[n];
150                    for (int i = 0; i < n; i++) {
151                        yhat[i] = getModelValue(getX(i), a);
152                    }
153                    return yhat;
154                }
155            };
156        }
157
158        public MultivariateMatrixFunction getModelFunctionJacobian() {
159            return new MultivariateMatrixFunction() {
160                public double[][] value(final double[] a)
161                    throws IllegalArgumentException {
162                    final int n = getNumObservations();
163                    final double[][] j = new double[n][];
164                    for (int i = 0; i < n; i++) {
165                        j[i] = getModelDerivatives(getX(i), a);
166                    }
167                    return j;
168                }
169            };
170        }
171    }

174     * Returns the name of this dataset.
175     *

179        return name;
180    }

182    /**
183     * Returns the total number of observations (data points).
184     *
185     * @return the number of observations
186     */
187    public int getNumObservations() {
188        return numObservations;
189    }
190
191    /**
192     * Returns a copy of the data arrays. The data is laid out as follows <li>
193     * {@code data[0][i] = x[i]},</li> <li>{@code data[1][i] = y[i]},</li>
194     *
195     * @return the array of data points.
196     */
197    public double[][] getData() {
198        return new double[][] {
199            MathArrays.copyOf(x), MathArrays.copyOf(y)
200        };
201    }
202
203    /**
204     * Returns the x-value of the {@code i}-th data point.
205     *
206     * @param i the index of the data point
207     * @return the x-value
208     */
209    public double getX(final int i) {
210        return x[i];
211    }
212
213    /**
214     * Returns the y-value of the {@code i}-th data point.
215     *
216     * @param i the index of the data point
217     * @return the y-value
218     */
219    public double getY(final int i) {
220        return y[i];
221    }
222
223    /**
224     * Returns the total number of parameters.
225     *
226     * @return the number of parameters
227     */
228    public int getNumParameters() {
229        return numParameters;
230    }
231
232    /**
233     * Returns the certified values of the paramters.
234     *
235     * @return the values of the parameters
236     */
237    public double[] getParameters() {
238        return MathArrays.copyOf(a);
239    }
240

242     * Returns the certified value of the {@code i}-th parameter.
243     *
244     * @param i the index of the parameter
245     * @return the value of the parameter
246     */
247    public double getParameter(final int i) {
248        return a[i];

252     * Reurns the certified values of the standard deviations of the parameters.
253     *
254     * @return the standard deviations of the parameters

256    public double[] getParametersStandardDeviations() {
257        return MathArrays.copyOf(sigA);
258    }
259
260    /**
261     * Returns the certified value of the standard deviation of the {@code i}-th
262     * parameter.
263     *
264     * @param i the index of the parameter
265     * @return the standard deviation of the parameter
266     */
267    public double getParameterStandardDeviation(final int i) {
268        return sigA[i];
269    }
270
271    /**
272     * Returns the certified value of the residual sum of squares.
273     *
274     * @return the residual sum of squares
275     */
276    public double getResidualSumOfSquares() {
277        return residualSumOfSquares;
278    }
279
280    /**
281     * Returns the total number of starting points (initial guesses for the
282     * optimization process).
283     *
284     * @return the number of starting points

286    public int getNumStartingPoints() {
287        return numStartingPoints;
288    }
289
290    /**
291     * Returns the {@code i}-th set of initial values of the parameters.

293     * @param i the index of the starting point

296    public double[] getStartingPoint(final int i) {
297        return MathArrays.copyOf(startingValues[i]);
298    }

300    /**
301     * Returns the least-squares problem corresponding to fitting the model to
302     * the specified data.
303     *
304     * @return the least-squares problem
305     */
306    public LeastSquaresProblem getLeastSquaresProblem() {
307        return problem;
308    }

310    /**
311     * Returns the value of the model for the specified values of the predictor

318    public abstract double getModelValue(final double x, final double[] a);

320    /**
321     * Returns the values of the partial derivatives of the model with respect

328    public abstract double[] getModelDerivatives(final double x,
329                                                 final double[] a);

331    /**
332     * <p>
333     * Parses the specified text lines, and extracts the indices of the first
334     * and last lines of the data defined by the specified {@code key}. This key
335     * must be one of
336     * </p>
337     * <ul>
338     * <li>{@code "Starting Values"},</li>
339     * <li>{@code "Certified Values"},</li>
340     * <li>{@code "Data"}.</li>
341     * </ul>
342     * <p>
343     * In the NIST data files, the line indices are separated by the keywords
344     * {@code "lines"} and {@code "to"}.
345     * </p>
346     *
347     * @param lines the line of text to be parsed
348     * @return an array of two {@code int}s. First value is the index of the
349     *         first line, second value is the index of the last line.
350     *         {@code null} if the line could not be parsed.
351     */
352    private static int[] findLineNumbers(final String key,
353                                         final Iterable<String> lines) {
354        for (String text : lines) {
355            boolean flag = text.contains(key) && text.contains("lines") &&
356                           text.contains("to") && text.contains(")");
357            if (flag) {
358                final int[] numbers = new int[2];
359                final String from = text.substring(text.indexOf("lines") + 5,
360                                                   text.indexOf("to"));
361                numbers[0] = Integer.parseInt(from.trim());
362                final String to = text.substring(text.indexOf("to") + 2,
363                                                 text.indexOf(")"));
364                numbers[1] = Integer.parseInt(to.trim());
365                return numbers;
366            }
367        }
368        return null;
369    }
