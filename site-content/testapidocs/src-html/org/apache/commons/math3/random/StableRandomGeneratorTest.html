017package org.apache.commons.math3.random;
018
019import org.apache.commons.math3.exception.OutOfRangeException;
020import org.apache.commons.math3.stat.StatUtils;
021import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
022import org.junit.Assert;
023import org.junit.Test;
024
025/**
026 * The class <code>StableRandomGeneratorTest</code> contains tests for the class
027 * {@link StableRandomGenerator}
028 * 
029 */
030public class StableRandomGeneratorTest {
031
032    private RandomGenerator rg = new Well19937c(100);
033    private final static int sampleSize = 10000;
034
035    /**
036     * Run the double nextDouble() method test Due to leptokurtic property the
037     * acceptance range is widened.
038     * 
039     * TODO: verify that tolerance this wide is really OK
040     */
041    @Test
042    public void testNextDouble() {
043        StableRandomGenerator generator = new StableRandomGenerator(rg, 1.3,
044                0.1);
045        double[] sample = new double[2 * sampleSize];
046        for (int i = 0; i < sample.length; ++i) {
047            sample[i] = generator.nextNormalizedDouble();
048        }
049        Assert.assertEquals(0.0, StatUtils.mean(sample), 0.3);
050    }
051
052    /**
053     * If alpha = 2, than it must be Gaussian distribution
054     */
055    @Test
056    public void testGaussianCase() {
057        StableRandomGenerator generator = new StableRandomGenerator(rg, 2d, 0.0);
058
059        double[] sample = new double[sampleSize];
060        for (int i = 0; i < sample.length; ++i) {
061            sample[i] = generator.nextNormalizedDouble();
062        }
063        Assert.assertEquals(0.0, StatUtils.mean(sample), 0.02);
064        Assert.assertEquals(1.0, StatUtils.variance(sample), 0.02);
065    }
066
067    /**
068     * If alpha = 1, than it must be Cauchy distribution
069     */
070    @Test
071    public void testCauchyCase() {
072        StableRandomGenerator generator = new StableRandomGenerator(rg, 1d, 0.0);
073        DescriptiveStatistics summary = new DescriptiveStatistics();
074
075        for (int i = 0; i < sampleSize; ++i) {
076            double sample = generator.nextNormalizedDouble();
077            summary.addValue(sample);
078        }
079
080        // Standard Cauchy distribution should have zero median and mode
081        double median = summary.getPercentile(50);
082        Assert.assertEquals(0.0, median, 0.2);
083    }
084
085    /**
086     * Input parameter range tests
087     */
088    @Test
089    public void testAlphaRangeBelowZero() {
090        try {
091            new StableRandomGenerator(rg,
092                    -1.0, 0.0);
093            Assert.fail("Expected OutOfRangeException");
094        } catch (OutOfRangeException e) {
095            Assert.assertEquals(-1.0, e.getArgument());
096        }
097    }
098
099    @Test
100    public void testAlphaRangeAboveTwo() {
101        try {
102            new StableRandomGenerator(rg,
103                    3.0, 0.0);
104            Assert.fail("Expected OutOfRangeException");
105        } catch (OutOfRangeException e) {
106            Assert.assertEquals(3.0, e.getArgument());
107        }
108    }
109
110    @Test
111    public void testBetaRangeBelowMinusOne() {
112        try {
113            new StableRandomGenerator(rg,
114                    1.0, -2.0);
115            Assert.fail("Expected OutOfRangeException");
116        } catch (OutOfRangeException e) {
117            Assert.assertEquals(-2.0, e.getArgument());
118        }
119    }
120
121    @Test
122    public void testBetaRangeAboveOne() {
123        try {
124            new StableRandomGenerator(rg,
125                    1.0, 2.0);
126            Assert.fail("Expected OutOfRangeException");
127        } catch (OutOfRangeException e) {
128            Assert.assertEquals(2.0, e.getArgument());
129        }
130    }
131}
