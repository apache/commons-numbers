017package org.apache.commons.math3.geometry.euclidean.twod;

019import java.util.List;
020
021import org.apache.commons.math3.geometry.euclidean.oned.Euclidean1D;
022import org.apache.commons.math3.geometry.euclidean.oned.IntervalsSet;
023import org.apache.commons.math3.geometry.partitioning.RegionFactory;
024import org.junit.Assert;
025import org.junit.Test;

027public class SubLineTest {
028
029    @Test
030    public void testEndPoints() {
031        Vector2D p1 = new Vector2D(-1, -7);
032        Vector2D p2 = new Vector2D(7, -1);
033        Segment segment = new Segment(p1, p2, new Line(p1, p2, 1.0e-10));
034        SubLine sub = new SubLine(segment);
035        List<Segment> segments = sub.getSegments();
036        Assert.assertEquals(1, segments.size());
037        Assert.assertEquals(0.0, new Vector2D(-1, -7).distance(segments.get(0).getStart()), 1.0e-10);
038        Assert.assertEquals(0.0, new Vector2D( 7, -1).distance(segments.get(0).getEnd()), 1.0e-10);
039    }
040
041    @Test
042    public void testNoEndPoints() {
043        SubLine wholeLine = new Line(new Vector2D(-1, 7), new Vector2D(7, 1), 1.0e-10).wholeHyperplane();
044        List<Segment> segments = wholeLine.getSegments();
045        Assert.assertEquals(1, segments.size());
046        Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
047                          segments.get(0).getStart().getX() < 0);
048        Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
049                          segments.get(0).getStart().getY() > 0);
050        Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
051                          segments.get(0).getEnd().getX() > 0);
052        Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
053                          segments.get(0).getEnd().getY() < 0);
054    }
055
056    @Test
057    public void testNoSegments() {
058        SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
059                                    new RegionFactory<Euclidean1D>().getComplement(new IntervalsSet(1.0e-10)));
060        List<Segment> segments = empty.getSegments();
061        Assert.assertEquals(0, segments.size());
062    }
063
064    @Test
065    public void testSeveralSegments() {
066        SubLine twoSubs = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
067                                    new RegionFactory<Euclidean1D>().union(new IntervalsSet(1, 2, 1.0e-10),
068                                                                           new IntervalsSet(3, 4, 1.0e-10)));
069        List<Segment> segments = twoSubs.getSegments();
070        Assert.assertEquals(2, segments.size());
071    }
072
073    @Test
074    public void testHalfInfiniteNeg() {
075        SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
076                                    new IntervalsSet(Double.NEGATIVE_INFINITY, 0.0, 1.0e-10));
077        List<Segment> segments = empty.getSegments();
078        Assert.assertEquals(1, segments.size());
079        Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getX()) &&
080                          segments.get(0).getStart().getX() < 0);
081        Assert.assertTrue(Double.isInfinite(segments.get(0).getStart().getY()) &&
082                          segments.get(0).getStart().getY() < 0);
083        Assert.assertEquals(0.0, new Vector2D(3, -4).distance(segments.get(0).getEnd()), 1.0e-10);
084    }
085
086    @Test
087    public void testHalfInfinitePos() {
088        SubLine empty = new SubLine(new Line(new Vector2D(-1, -7), new Vector2D(7, -1), 1.0e-10),
089                                    new IntervalsSet(0.0, Double.POSITIVE_INFINITY, 1.0e-10));
090        List<Segment> segments = empty.getSegments();
091        Assert.assertEquals(1, segments.size());
092        Assert.assertEquals(0.0, new Vector2D(3, -4).distance(segments.get(0).getStart()), 1.0e-10);
093        Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getX()) &&
094                          segments.get(0).getEnd().getX() > 0);
095        Assert.assertTrue(Double.isInfinite(segments.get(0).getEnd().getY()) &&
096                          segments.get(0).getEnd().getY() > 0);
097    }

099    @Test
100    public void testIntersectionInsideInside() {
101        SubLine sub1 = new SubLine(new Vector2D(1, 1), new Vector2D(3, 1), 1.0e-10);
102        SubLine sub2 = new SubLine(new Vector2D(2, 0), new Vector2D(2, 2), 1.0e-10);
103        Assert.assertEquals(0.0, new Vector2D(2, 1).distance(sub1.intersection(sub2, true)),  1.0e-12);
104        Assert.assertEquals(0.0, new Vector2D(2, 1).distance(sub1.intersection(sub2, false)), 1.0e-12);
105    }
