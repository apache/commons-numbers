064    // Generic test function.  Takes params x and y and tests them for
065    // equality.  Then checks the status flags against the flags argument.
066    // If the test fail, it prints the desc string
067    private void test(Dfp x, Dfp y, int flags, String desc)
068    {
069        boolean b = x.equals(y);
070
071        if (!x.equals(y) && !x.unequal(y))  // NaNs involved
072            b = (x.toString().equals(y.toString()));
073
074        if (x.equals(field.newDfp("0")))  // distinguish +/- zero
075            b = (b && (x.toString().equals(y.toString())));
076
077        b = (b && x.getField().getIEEEFlags() == flags);
078
079        if (!b)
080            Assert.assertTrue("assersion failed "+desc+" x = "+x.toString()+" flags = "+x.getField().getIEEEFlags(), b);
081
082        x.getField().clearIEEEFlags();
083    }

373    // utility function to help test comparisons
374    private void cmptst(Dfp a, Dfp b, String op, boolean result, double num)
375    {
376        if (op == "equal")
377            if (a.equals(b) != result)
378                Assert.fail("assersion failed.  "+op+" compare #"+num);
379
380        if (op == "unequal")
381            if (a.unequal(b) != result)
382                Assert.fail("assersion failed.  "+op+" compare #"+num);
383
384        if (op == "lessThan")
385            if (a.lessThan(b) != result)
386                Assert.fail("assersion failed.  "+op+" compare #"+num);
387
388        if (op == "greaterThan")
389            if (a.greaterThan(b) != result)
390                Assert.fail("assersion failed.  "+op+" compare #"+num);
391    }
