017package org.apache.commons.math3.distribution;
018
019import org.apache.commons.math3.exception.DimensionMismatchException;
020import org.apache.commons.math3.exception.MathArithmeticException;
021import org.apache.commons.math3.exception.NotANumberException;
022import org.apache.commons.math3.exception.NotFiniteNumberException;
023import org.apache.commons.math3.exception.NotPositiveException;
024import org.apache.commons.math3.util.FastMath;
025import org.junit.Assert;
026import org.junit.Test;
027
028/**
029 * Test class for {@link EnumeratedIntegerDistribution}.
030 *
031 */
032public class EnumeratedIntegerDistributionTest {
033
034    /**
035     * The distribution object used for testing.
036     */
037    private final EnumeratedIntegerDistribution testDistribution;
038
039    /**
040     * Creates the default distribution object used for testing.
041     */
042    public EnumeratedIntegerDistributionTest() {
043        // Non-sorted singleton array with duplicates should be allowed.
044        // Values with zero-probability do not extend the support.
045        testDistribution = new EnumeratedIntegerDistribution(
046                new int[]{3, -1, 3, 7, -2, 8},
047                new double[]{0.2, 0.2, 0.3, 0.3, 0.0, 0.0});
048    }
049
050    /**
051     * Tests if the EnumeratedIntegerDistribution constructor throws
052     * exceptions for invalid data.
053     */
054    @Test
055    public void testExceptions() {
056        EnumeratedIntegerDistribution invalid = null;
057        try {
058            new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0});
059            Assert.fail("Expected DimensionMismatchException");
060        } catch (DimensionMismatchException e) {
061        }
062        try {
063            new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, -1.0});
064            Assert.fail("Expected NotPositiveException");
065        } catch (NotPositiveException e) {
066        }
067        try {
068            new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, 0.0});
069            Assert.fail("Expected MathArithmeticException");
070        } catch (MathArithmeticException e) {
071        }
072        try {
073          new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, Double.NaN});
074            Assert.fail("Expected NotANumberException");
075        } catch (NotANumberException e) {
076        }
077        try {
078        new EnumeratedIntegerDistribution(new int[]{1, 2}, new double[]{0.0, Double.POSITIVE_INFINITY});
079            Assert.fail("Expected NotFiniteNumberException");
080        } catch (NotFiniteNumberException e) {
081        }
082        Assert.assertNull("Expected non-initialized DiscreteRealDistribution", invalid);
083    }
084
085    /**
086     * Tests if the distribution returns proper probability values.
087     */
088    @Test
089    public void testProbability() {
090        int[] points = new int[]{-2, -1, 0, 1, 2, 3, 4, 5, 6, 7, 8};
091        double[] results = new double[]{0, 0.2, 0, 0, 0, 0.5, 0, 0, 0, 0.3, 0};
092        for (int p = 0; p < points.length; p++) {
093            double probability = testDistribution.probability(points[p]);
094            Assert.assertEquals(results[p], probability, 0.0);
095        }
096    }
097
098    /**
099     * Tests if the distribution returns proper cumulative probability values.
100     */
101    @Test
102    public void testCumulativeProbability() {
103        int[] points = new int[]{-2, -1, 0, 1, 2, 3, 4, 5, 6, 7, 8};
104        double[] results = new double[]{0, 0.2, 0.2, 0.2, 0.2, 0.7, 0.7, 0.7, 0.7, 1.0, 1.0};
105        for (int p = 0; p < points.length; p++) {
106            double probability = testDistribution.cumulativeProbability(points[p]);
107            Assert.assertEquals(results[p], probability, 1e-10);
108        }
109    }
110
111    /**
112     * Tests if the distribution returns proper mean value.
113     */
114    @Test
115    public void testGetNumericalMean() {
116        Assert.assertEquals(3.4, testDistribution.getNumericalMean(), 1e-10);
117    }
118
119    /**
120     * Tests if the distribution returns proper variance.
121     */
122    @Test
123    public void testGetNumericalVariance() {
124        Assert.assertEquals(7.84, testDistribution.getNumericalVariance(), 1e-10);
125    }
126
127    /**
128     * Tests if the distribution returns proper lower bound.
129     */
130    @Test
131    public void testGetSupportLowerBound() {
132        Assert.assertEquals(-1, testDistribution.getSupportLowerBound());
133    }
134
135    /**
136     * Tests if the distribution returns proper upper bound.
137     */
138    @Test
139    public void testGetSupportUpperBound() {
140        Assert.assertEquals(7, testDistribution.getSupportUpperBound());
141    }
142
143    /**
144     * Tests if the distribution returns properly that the support is connected.
145     */
146    @Test
147    public void testIsSupportConnected() {
148        Assert.assertTrue(testDistribution.isSupportConnected());
149    }
150
151    /**
152     * Tests sampling.
153     */
154    @Test
155    public void testSample() {
156        final int n = 1000000;
157        testDistribution.reseedRandomGenerator(-334759360); // fixed seed
158        final int[] samples = testDistribution.sample(n);
159        Assert.assertEquals(n, samples.length);
160        double sum = 0;
161        double sumOfSquares = 0;
162        for (int i = 0; i < samples.length; i++) {
163            sum += samples[i];
164            sumOfSquares += samples[i] * samples[i];
165        }
166        Assert.assertEquals(testDistribution.getNumericalMean(),
167                sum / n, 1e-2);
168        Assert.assertEquals(testDistribution.getNumericalVariance(),
169                sumOfSquares / n - FastMath.pow(sum / n, 2), 1e-2);
170    }
171}
