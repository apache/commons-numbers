018package org.apache.commons.math3.optimization;
019
020
021import org.apache.commons.math3.analysis.MultivariateFunction;
022import org.apache.commons.math3.optimization.direct.NelderMeadSimplex;
023import org.apache.commons.math3.optimization.direct.SimplexOptimizer;
024import org.apache.commons.math3.random.GaussianRandomGenerator;
025import org.apache.commons.math3.random.JDKRandomGenerator;
026import org.apache.commons.math3.random.RandomVectorGenerator;
027import org.apache.commons.math3.random.UncorrelatedRandomVectorGenerator;
028import org.junit.Assert;
029import org.junit.Test;
030
031@Deprecated
032public class MultivariateMultiStartOptimizerTest {
033    @Test
034    public void testRosenbrock() {
035        Rosenbrock rosenbrock = new Rosenbrock();
036        SimplexOptimizer underlying
037            = new SimplexOptimizer(new SimpleValueChecker(-1, 1.0e-3));
038        NelderMeadSimplex simplex = new NelderMeadSimplex(new double[][] {
039                { -1.2,  1.0 }, { 0.9, 1.2 } , {  3.5, -2.3 }
040            });
041        underlying.setSimplex(simplex);
042        JDKRandomGenerator g = new JDKRandomGenerator();
043        g.setSeed(16069223052l);
044        RandomVectorGenerator generator =
045            new UncorrelatedRandomVectorGenerator(2, new GaussianRandomGenerator(g));
046        MultivariateMultiStartOptimizer optimizer =
047            new MultivariateMultiStartOptimizer(underlying, 10, generator);
048        PointValuePair optimum =
049            optimizer.optimize(1100, rosenbrock, GoalType.MINIMIZE, new double[] { -1.2, 1.0 });
050
051        Assert.assertEquals(rosenbrock.getCount(), optimizer.getEvaluations());
052        Assert.assertTrue(optimizer.getEvaluations() > 900);
053        Assert.assertTrue(optimizer.getEvaluations() < 1200);
054        Assert.assertTrue(optimum.getValue() < 8.0e-4);
055    }
056
057    private static class Rosenbrock implements MultivariateFunction {
058        private int count;
059
060        public Rosenbrock() {
061            count = 0;
062        }
063
064        public double value(double[] x) {
065            ++count;
066            double a = x[1] - x[0] * x[0];
067            double b = 1.0 - x[0];
068            return 100 * a * a + b * b;
069        }
070
071        public int getCount() {
072            return count;
073        }
074    }
075}
