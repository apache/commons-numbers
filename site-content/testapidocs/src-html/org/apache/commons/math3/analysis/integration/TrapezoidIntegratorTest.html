017package org.apache.commons.math3.analysis.integration;
018
019import org.apache.commons.math3.analysis.QuinticFunction;
020import org.apache.commons.math3.analysis.UnivariateFunction;
021import org.apache.commons.math3.analysis.function.Sin;
022import org.apache.commons.math3.exception.NumberIsTooLargeException;
023import org.apache.commons.math3.exception.NumberIsTooSmallException;
024import org.apache.commons.math3.util.FastMath;
025import org.junit.Assert;
026import org.junit.Test;
027
028
029/**
030 * Test case for trapezoid integrator.
031 * <p>
032 * Test runs show that for a default relative accuracy of 1E-6, it
033 * generally takes 10 to 15 iterations for the integral to converge.
034 *
035 */
036public final class TrapezoidIntegratorTest {
037
038    /**
039     * Test of integrator for the sine function.
040     */
041    @Test
042    public void testSinFunction() {
043        UnivariateFunction f = new Sin();
044        UnivariateIntegrator integrator = new TrapezoidIntegrator();
045        double min, max, expected, result, tolerance;
046
047        min = 0; max = FastMath.PI; expected = 2;
048        tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
049        result = integrator.integrate(10000, f, min, max);
050        Assert.assertTrue(integrator.getEvaluations() < 2500);
051        Assert.assertTrue(integrator.getIterations()  < 15);
052        Assert.assertEquals(expected, result, tolerance);
053
054        min = -FastMath.PI/3; max = 0; expected = -0.5;
055        tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
056        result = integrator.integrate(10000, f, min, max);
057        Assert.assertTrue(integrator.getEvaluations() < 2500);
058        Assert.assertTrue(integrator.getIterations()  < 15);
059        Assert.assertEquals(expected, result, tolerance);
060    }
061
062    /**
063     * Test of integrator for the quintic function.
064     */
065    @Test
066    public void testQuinticFunction() {
067        UnivariateFunction f = new QuinticFunction();
068        UnivariateIntegrator integrator = new TrapezoidIntegrator();
069        double min, max, expected, result, tolerance;
070
071        min = 0; max = 1; expected = -1.0/48;
072        tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
073        result = integrator.integrate(10000, f, min, max);
074        Assert.assertTrue(integrator.getEvaluations() < 5000);
075        Assert.assertTrue(integrator.getIterations()  < 15);
076        Assert.assertEquals(expected, result, tolerance);
077
078        min = 0; max = 0.5; expected = 11.0/768;
079        tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
080        result = integrator.integrate(10000, f, min, max);
081        Assert.assertTrue(integrator.getEvaluations() < 2500);
082        Assert.assertTrue(integrator.getIterations()  < 15);
083        Assert.assertEquals(expected, result, tolerance);
084
085        min = -1; max = 4; expected = 2048/3.0 - 78 + 1.0/48;
086        tolerance = FastMath.abs(expected * integrator.getRelativeAccuracy());
087        result = integrator.integrate(10000, f, min, max);
088        Assert.assertTrue(integrator.getEvaluations() < 5000);
089        Assert.assertTrue(integrator.getIterations()  < 15);
090        Assert.assertEquals(expected, result, tolerance);
091
092    }
093
094    /**
095     * Test of parameters for the integrator.
096     */
097    @Test
098    public void testParameters() {
099        UnivariateFunction f = new Sin();
100
101        try {
102            // bad interval
103            new TrapezoidIntegrator().integrate(1000, f, 1, -1);
104            Assert.fail("Expecting NumberIsTooLargeException - bad interval");
105        } catch (NumberIsTooLargeException ex) {
106            // expected
107        }
108        try {
109            // bad iteration limits
110            new TrapezoidIntegrator(5, 4);
111            Assert.fail("Expecting NumberIsTooSmallException - bad iteration limits");
112        } catch (NumberIsTooSmallException ex) {
113            // expected
114        }
115        try {
116            // bad iteration limits
117            new TrapezoidIntegrator(10,99);
118            Assert.fail("Expecting NumberIsTooLargeException - bad iteration limits");
119        } catch (NumberIsTooLargeException ex) {
120            // expected
121        }
122    }
